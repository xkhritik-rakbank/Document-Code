public static File findXMLFile(String folderPath, String targetFileName) {
    File folder = new File(folderPath);
    
    // Look for the specific XML file matching the target name
    File targetFile = new File(folder, targetFileName + ".xml");
    
    // Check if the target file exists and return it
    if (targetFile.exists() && targetFile.isFile()) {
        return targetFile;
    }
    
    // If target file not found, return null
    return null;
}
