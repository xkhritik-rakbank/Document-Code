// Check the format of your cacertsFile
try {
    KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
    FileInputStream fis = new FileInputStream(cacertsFile);
    ks.load(fis, SSL_password.toCharArray());
    fis.close();
    System.out.println("Successfully loaded keystore: " + cacertsFile);
} catch (Exception e) {
    System.err.println("Error loading keystore: " + e.getMessage());
    e.printStackTrace();
}


// Only set the necessary SSL properties
System.setProperty("javax.net.ssl.trustStore", cacertsFile);
System.setProperty("javax.net.ssl.trustStorePassword", SSL_password);
System.setProperty("javax.net.ssl.trustStoreType", KeyStore.getDefaultType());

// If you need client authentication:
System.setProperty("javax.net.ssl.keyStore", cacertsFile);
System.setProperty("javax.net.ssl.keyStorePassword", SSL_password);
System.setProperty("javax.net.ssl.keyStoreType", KeyStore.getDefaultType());
