@echo off
:start
cls

REM Compile Java files (if needed)
javac YourMainClass.java

REM Run the Java application
java YourMainClass

REM Prompt for continuation
echo.
echo Do you want to run the application again?
set /p choice="Press 'Y' to continue, or any other key to exit: "

REM Check user input
if /I "%choice%"=="Y" goto start

REM Optional: Add a pause before completely closing
pause
