import com.aspose.words.*;
 
public class AddImageFooterToDocument {
    public static void main(String[] args) {
        try {
            // Load an existing document or create a new one
            Document doc = new Document(); // Use new Document() for a new document
            DocumentBuilder builder = new DocumentBuilder(doc);
 
            // Move to the footer of the document (primary footer for all pages)
            builder.moveToHeaderFooter(HeaderFooterType.FOOTER_PRIMARY);
 
            // Insert the image as the footer
            String imagePath = "C:\\Users\\varun-k\\Documents\\footer_image.jpg";
            builder.insertImage(imagePath, RelativeHorizontalPosition.PAGE, 0, 
                    RelativeVerticalPosition.PAGE, 740,  // Adjust vertical position based on page size
                    100, 50,  // Set width and height of the image
                    WrapType.NONE);  // No text wrapping, image is placed at fixed coordinates
 
            // Return to the main body of the document
            builder.moveToDocumentEnd();
 
            // Insert some content in the main body of the document (optional)
            builder.writeln("This is the main content of the document.");
 
            // Save the document with the footer image as DOCX
            doc.save("C:\\Users\\varun-k\\Documents\\document_with_image_footer.docx");
 
            // Convert the document to PDF with the image footer
            doc.save("C:\\Users\\varun-k\\Documents\\document_with_image_footer.pdf", SaveFormat.PDF);
 
            System.out.println("Image footer added and document converted to PDF successfully!");
 
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
